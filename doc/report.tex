\subsection{If then else}

\begin{itemize}

\BinaryInfC{$\Phi_1 \vdash$ \textbf{if} c \textbf{then} t \textbf{else} e $\stepsto$ $\Phi_3 \vdash$ v}
\eproof

\item

\sproof

\eproof

\end{itemize}
